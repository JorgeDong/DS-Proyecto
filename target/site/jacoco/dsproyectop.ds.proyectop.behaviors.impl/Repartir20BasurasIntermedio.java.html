        for (int i = 0; i < veinte; i++) {
            combinacionesTotal.add("id" + i);
        }
        for (int i = 0; i < veinte; i++) {
            this.iteracionesTotal.add(
            this.combinacionesTotal.get(i));
            this.iteracionesTotal.add(
            this.combinacionesTotal.get(i));
        }
        return this.iteracionesTotal;
